#GgjReviewBox

## Basic usage

```
import GgjReviewBox, { IGgjReviewBox } from 'components/commons/GgjReviewBox'

const mockData: IGgjReviewBox = {
   {
      masterId: 1,
      comment: '',
      userName: '',
      updatedAt: '',
      rating: 4.5,
    },
  }

<GgjReviewBox data={mockData} />
```
